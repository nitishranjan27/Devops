----Create a database-------

create database WorkLineDB

use WorkLineDB

----Create a Table-----

create table Users(
MemberId int not null primary key,
MemberName varchar(max) not null,
EffectiveDate Date not null,
DateOfJoining Date not null,
Unit varchar(max) not null,
Business varchar(max) not null,
Capability varchar(max) not null,
Team varchar(max) not null,
Designation varchar(max) not null,
Grade varchar(max),
County varchar(max) not null,
City varchar(max) not null,
OfficeName varchar(max) not null,
State varchar(max),
LocationType varchar(max) not null,
FunctionalAppriserID int not null,
FunctionalAppriserName varchar(max) not null,
SkipLevelManagerID int not null,
SkipLevelManagerName varchar(max) not null,
SLACode varchar(max) not null,
CostType varchar(max) not null,
Percentage int not null,
BillableType varchar(max) not null,
Remark varchar(max),
EmployeeCount int not null,
BillingModel varchar(max) not null,
ClientBrand varchar(max)
);

select * from Users;
